import React from 'react';

  totalLojas = 41,

      minimumFractionDigits: 3,
      maximumFractionDigits: 3,

          />
        </svg>
        <span className="text-xl font-bold text-gray-800">{rotaName}</span>
        <span className="text-xl text-gray-500">- {totalLojas} {tipoLoja}</span>
      </div>

      {/* Content */}
      <div className="flex items-center gap-8">
        {/* Donut Chart */}
        <div className="relative w-36 h-36 flex-shrink-0">
          <svg className="w-full h-full transform -rotate-90" viewBox="0 0 120 120">
            {/* Background circle (red) */}
            <circle
              cx="60"
              cy="60"
              r="54"
              fill="none"
              stroke="#DC2626"
              strokeWidth="12"
            />
            {/* Progress circle (green) */}
            <circle
              cx="60"
              cy="60"
              r="54"
              fill="none"
              stroke="#16A34A"
              strokeWidth="12"
              strokeDasharray={circumference}
              strokeDashoffset={greenOffset}
              strokeLinecap="butt"
            />
          </svg>
          {/* Center text */}
          <div className="absolute inset-0 flex flex-col items-center justify-center">
            <span className="text-3xl font-bold text-gray-900">{atingimento}%</span>
            <span className="text-sm text-gray-600">Atingimento</span>
          </div>
        </div>

        {/* Stats */}
        <div className="flex-1">
          {/* Saldo */}
          <div className="mb-4">
            <div className="text-3xl font-bold text-green-600">
              R$ {formatCurrency(saldo)}
            </div>
            <div className="text-gray-600 text-sm">Saldo</div>
            <div className="text-gray-500 text-sm">Meta: R$ {formatCurrency(meta)}</div>
          </div>

          {/* Divider */}
          <div className="border-t border-gray-200 my-4"></div>

          {/* Metrics row */}
          <div className="flex justify-between text-center">
            {/* Lojas */}
            <div className="flex flex-col items-center">
              <div className="flex items-center gap-1">
                <svg 
                  className="w-4 h-4 text-gray-500" 
                  fill="none" 
                  stroke="currentColor" 
                  viewBox="0 0 24 24"
                >
                  <path

                  />
                </svg>
                <span className="font-bold text-gray-800">{totalLojas}</span>
              </div>
              <span className="text-xs text-gray-500">Lojas</span>
            </div>

            {/* Cidades */}
            <div className="flex flex-col items-center">
              <div className="flex items-center gap-1">
                <svg 
                  className="w-4 h-4 text-gray-500" 
                  fill="none" 
                  stroke="currentColor" 
                  viewBox="0 0 24 24"
                >
                  <path 
                    strokeLinecap="round" 
                    strokeLinejoin="round" 
                    strokeWidth={1.5} 
                    d="M17.657 16.657L13.414 20.9a1.998 1.998 0 01-2.827 0l-4.244-4.243a8 8 0 1111.314 0z" 
                  />
                  <path 
                    strokeLinecap="round" 
                    strokeLinejoin="round" 
                    strokeWidth={1.5}

          </div>

      </div>

        </span>

    </div>

export default RotaCard;
